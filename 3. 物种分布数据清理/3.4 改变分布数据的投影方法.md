## 分布数据重新投影

```r
library(rgdal)
library(maptools)
library(raster)
library(tidyverse)
setwd("E:/sjdata/")
data = read.csv("./aa.csv") %>% data.frame(.)
names(data)[2:3] <- c("long","lat")
cord.dec = SpatialPoints(cbind(data$long, -data$lat), proj4string = CRS("+proj=longlat"))
head(data)

# Setting existing coordinate as lat-long system
cord.dec = SpatialPoints(cbind(data$long, data$lat), proj4string = CRS("+proj=longlat +ellps=WGS84 +datum=WGS84")  )
cord.dec@coords
# Transforming coordinate to UTM using EPSG=32748 for WGS=84, UTM Zone=48M,
# Southern Hemisphere)
cord.UTM <- spTransform(cord.dec, CRS("+init=epsg:32647"))
write.csv(cord.UTM@coords,"C:/Users/admin/Desktop/hs2.csv")

```
